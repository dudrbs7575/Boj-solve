#include <bits/stdc++.h>

using namespace std;

bool chk[200000001];

int solution(int n, vector<int> stations, int w)
{
    int answer = 0;
    for(int i=0;i<stations.size();i++){
        int cur=stations[i];
        chk[cur]=true;
        for(int j=cur-1;j>=cur-w;j--){
            if(j<0)
                break;
            chk[j]=true;
        }
        for(int j=cur+1;j<=cur+w;j++){
            if(j>n)
                break;
            chk[j]=true;
        }
    }
    
    int cnt=0;
    for(int i=1;i<=n;){
        if(!chk[i]){
            if(cnt==w){
                answer++;
                i+=(w+1);
                cnt=0;
                continue;
            }
            cnt++;
        }
        else{
            if(cnt!=0){
                answer++;
                cnt=0;
            }
        }
        i++;
    }
    if(cnt!=0)
        answer++;
    
    return answer;
}
